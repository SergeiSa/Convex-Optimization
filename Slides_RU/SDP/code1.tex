n = 7; A = randn(n, n) - 3*rand*eye(n);
Q = eye(n); I = eye(n); epsilon = 0.001;

cvx_begin sdp
    variable P(n, n) symmetric
    minimize 0
    subject to
        P >= epsilon*I;
        A'*P + P*A + Q <= 0;
cvx_end

if strcmp(cvx_status, 'Solved')
    [eig(A), eig(A*P + P*A' + Q), eig(P)]
else
    eig(A)
end
