\begin{lstlisting}[language=Matlab]
n = 5; m = 2;
A = randn(n, n);
B = randn(n, m);
Q = eye(n)*0.1;  I = eye(n); epsilon = 0.001;
cvx_begin sdp
    variable P(n, n) symmetric
    variable L(m, n)
    
    minimize 0
    subject to
    	P >= epsilon*I;
        A*P + P*A' + B*L + L'*B' + Q <= epsilon*I;
cvx_end
P = full(P);
L = full(L);
K = L*pinv(P);

disp('CL eig:')
eig(A + B*K)
\end{lstlisting}
