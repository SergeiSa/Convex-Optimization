Q = eye(n); I = eye(n); epsilon = 0.001;

cvx_begin sdp
    variable P(n, n) symmetric
    minimize 0
    subject to
        P >= epsilon*I;
        A'*P*A - P + Q <= 0;
cvx_end

if strcmp(cvx_status, 'Solved')
    [abs(eig(A)), eig(A'*P*A - P), eig(P)]
else
    abs(eig(A))
end
